添加 CSS 樣式
在 monthly_report.php 的 CSS 樣式中添加以下通用固定樣式：
CSS - 通用表格固定樣式
/* ========== 電腦與手機通用表格固定 ========== */ /* 這些樣式會被 JavaScript 動態應用 */ .monthly-table.js-fixed thead { position: sticky !important; top: 0 !important; z-index: 100 !important; background: #f8fafc !important; backdrop-filter: blur(10px) !important; -webkit-backdrop-filter: blur(10px) !important; } .monthly-table.js-fixed tbody td:first-child { position: sticky !important; left: 0 !important; z-index: 90 !important; background: #f8fafc !important; backdrop-filter: blur(10px) !important; -webkit-backdrop-filter: blur(10px) !important; box-shadow: 2px 0 8px rgba(0,0,0,0.1) !important; } /* 手機專用優化 */ @media (max-width: 768px) { .monthly-table.js-fixed { font-size: 14px !important; } .monthly-table.js-fixed thead { top: 0 !important; } .monthly-table.js-fixed tbody td:first-child { left: 0 !important; min-width: 140px !important; } .monthly-table.js-fixed th, .monthly-table.js-fixed td { padding: 10px 6px !important; } }
2
添加 JavaScript 程式碼
在 monthly_report.php 的 </body> 標籤前添加以下 JavaScript：
JavaScript - 通用表格固定
<script> // 電腦與手機通用表格固定方案 document.addEventListener('DOMContentLoaded', function() { const tableContainer = document.querySelector('.table-container'); const monthlyTable = document.querySelector('.monthly-table'); if (!tableContainer || !monthlyTable) return; // 添加 JS 固定類別 monthlyTable.classList.add('js-fixed'); const thead = monthlyTable.querySelector('thead'); const firstCells = monthlyTable.querySelectorAll('tbody td:first-child'); if (!thead || firstCells.length === 0) return; // 監聽滾動事件 let isScrolling = false; let animationFrameId = null; tableContainer.addEventListener('scroll', function() { if (!isScrolling) { isScrolling = true; // 使用 requestAnimationFrame 優化性能 animationFrameId = requestAnimationFrame(() => { const scrollLeft = this.scrollLeft; const scrollTop = this.scrollTop; // 動態調整表頭位置 thead.style.transform = `translateY(${scrollTop}px)`; // 動態調整左欄位置 firstCells.forEach(cell => { cell.style.transform = `translateX(${scrollLeft}px)`; }); isScrolling = false; }); } }); // 初始調整（觸發一次滾動以啟用固定） setTimeout(() => { tableContainer.scrollLeft = 1; tableContainer.scrollTop = 1; setTimeout(() => { tableContainer.scrollLeft = 0; tableContainer.scrollTop = 0; }, 50); }, 500); // 手機專用優化 if (window.innerWidth <= 768) { monthlyTable.style.fontSize = '14px'; const cells = monthlyTable.querySelectorAll('th, td'); cells.forEach(cell => { cell.style.padding = '10px 6px'; }); firstCells.forEach(cell => { cell.style.minWidth = '140px'; }); } // 監聽視窗大小變化（響應式設計） window.addEventListener('resize', function() { if (window.innerWidth <= 768) { monthlyTable.style.fontSize = '14px'; const cells = monthlyTable.querySelectorAll('th, td'); cells.forEach(cell => { cell.style.padding = '10px 6px'; }); firstCells.forEach(cell => { cell.style.minWidth = '140px'; }); } else { monthlyTable.style.fontSize = ''; const cells = monthlyTable.querySelectorAll('th, td'); cells.forEach(cell => { cell.style.padding = ''; }); firstCells.forEach(cell => { cell.style.minWidth = ''; }); } }); console.log('✅ 電腦與手機通用表格固定已啟用'); }); </script>
